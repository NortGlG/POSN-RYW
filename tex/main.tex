\begin{frame}{Arithmetic Operators Precedence}
    \centering
    \begin{tabular}{|c|c|c|c|}
        \hline
        Order & Operator     & Associativity Type \\
        \hline
        1     & ()           & -                  \\
        \hline
        2     & **           & right to left      \\
        \hline
        3     & *, /, //, \% & left to right      \\
        \hline
        4     & +, -         & left to right      \\
        \hline
    \end{tabular}
\end{frame}
